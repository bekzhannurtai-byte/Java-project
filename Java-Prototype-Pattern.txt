import java.util.ArrayList;
import java.util.List;

interface OrderPrototype {
    OrderPrototype clone();
}

class Order implements OrderPrototype {
    private String clientName;
    private List<String> products;
    private String deliveryAddress;
    private String deliveryType;
    private double discount;
    private String paymentMethod;
    private String paymentStatus;
    private String contactPhone;

    public Order(String clientName, List<String> products, String deliveryAddress, 
                 String deliveryType, double discount, String paymentMethod, 
                 String paymentStatus, String contactPhone) {
        this.clientName = clientName;
        this.products = new ArrayList<>(products);
        this.deliveryAddress = deliveryAddress;
        this.deliveryType = deliveryType;
        this.discount = discount;
        this.paymentMethod = paymentMethod;
        this.paymentStatus = paymentStatus;
        this.contactPhone = contactPhone;
    }

    @Override
    public Order clone() {
        return new Order(this.clientName, this.products, this.deliveryAddress, 
                         this.deliveryType, this.discount, this.paymentMethod, 
                         this.paymentStatus, this.contactPhone);
    }

    public void setDeliveryAddress(String deliveryAddress) {
        this.deliveryAddress = deliveryAddress;
    }

    public void setPaymentStatus(String paymentStatus) {
        this.paymentStatus = paymentStatus;
    }

    public void showInfo() {
        System.out.println("==========================================");
        System.out.println("КЛИЕНТ: " + clientName + " | ТЕЛ: " + contactPhone);
        System.out.println("ТОВАРЫ: " + products);
        System.out.println("ДОСТАВКА: " + deliveryType + " | АДРЕС: " + deliveryAddress);
        System.out.println("ОПЛАТА: " + paymentMethod + " | СТАТУС: " + paymentStatus);
        System.out.println("СКИДКА: " + discount + "%");
    }
}
